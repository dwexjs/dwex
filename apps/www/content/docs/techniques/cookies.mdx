## Overview

Dwex provides utilities for working with HTTP cookies for session management and authentication.

## Example

```typescript
@Controller("auth")
export class AuthController {
	@Get("login")
	login(@Res() res: Response) {
		res.cookie("session", "token", {
			httpOnly: true,
			secure: true,
			maxAge: 3600000,
		});
		return res.json({ success: true });
	}
}
```
