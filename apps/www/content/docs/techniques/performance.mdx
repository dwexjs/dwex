## Example: Caching

```typescript
@Injectable()
export class UserService {
	private cache = new Map();

	async findOne(id: string) {
		if (this.cache.has(id)) {
			return this.cache.get(id);
		}

		const user = await this.db.findOne(id);
		this.cache.set(id, user);
		return user;
	}
}
```
